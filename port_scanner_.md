import socket
import threading
from datetime import datetime

class PortScanner:
    def __init__(self, target, start_port=1, end_port=1000, max_threads=100):
        self.target = target
        self.start_port = start_port
        self.end_port = end_port
        self.max_threads = max_threads
        self.open_ports = []
        self.lock = threading.Lock()
    
    def scan_port(self, port):
        """–°–∫–∞–Ω–∏—Ä–æ–≤–∞–Ω–∏–µ –æ–¥–Ω–æ–≥–æ –ø–æ—Ä—Ç–∞"""
        try:
            # –°–æ–∑–¥–∞–µ–º —Å–æ–∫–µ—Ç
            sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
            sock.settimeout(1)
            
            # –ü—ã—Ç–∞–µ–º—Å—è –ø–æ–¥–∫–ª—é—á–∏—Ç—å—Å—è
            result = sock.connect_ex((self.target, port))
            
            if result == 0:
                with self.lock:
                    self.open_ports.append(port)
                    print(f"–ü–æ—Ä—Ç {port}: –û–¢–ö–†–´–¢")
            
            sock.close()
            
        except Exception as e:
            pass
    
    def get_service_name(self, port):
        """–ü–æ–ª—É—á–µ–Ω–∏–µ –∏–º–µ–Ω–∏ —Å–ª—É–∂–±—ã –¥–ª—è –ø–æ—Ä—Ç–∞"""
        try:
            return socket.getservbyport(port)
        except:
            return "–Ω–µ–∏–∑–≤–µ—Å—Ç–Ω–æ"
    
    def run_scan(self):
        """–ó–∞–ø—É—Å–∫ —Å–∫–∞–Ω–∏—Ä–æ–≤–∞–Ω–∏—è"""
        print(f"–ù–∞—á–∞–ª–æ —Å–∫–∞–Ω–∏—Ä–æ–≤–∞–Ω–∏—è {self.target}")
        print(f"–î–∏–∞–ø–∞–∑–æ–Ω –ø–æ—Ä—Ç–æ–≤: {self.start_port}-{self.end_port}")
        print("-" * 50)
        
        start_time = datetime.now()
        threads = []
        
        # –°–æ–∑–¥–∞–µ–º –ø–æ—Ç–æ–∫–∏ –¥–ª—è —Å–∫–∞–Ω–∏—Ä–æ–≤–∞–Ω–∏—è
        for port in range(self.start_port, self.end_port + 1):
            thread = threading.Thread(target=self.scan_port, args=(port,))
            threads.append(thread)
            thread.start()
            
            # –û–≥—Ä–∞–Ω–∏—á–∏–≤–∞–µ–º –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –æ–¥–Ω–æ–≤—Ä–µ–º–µ–Ω–Ω–æ —Ä–∞–±–æ—Ç–∞—é—â–∏—Ö –ø–æ—Ç–æ–∫–æ–≤
            if len(threads) >= self.max_threads:
                for t in threads:
                    t.join()
                threads = []
        
        # –ñ–¥–µ–º –∑–∞–≤–µ—Ä—à–µ–Ω–∏—è –æ—Å—Ç–∞–≤—à–∏—Ö—Å—è –ø–æ—Ç–æ–∫–æ–≤
        for thread in threads:
            thread.join()
        
        end_time = datetime.now()
        
        # –í—ã–≤–æ–¥ —Ä–µ–∑—É–ª—å—Ç–∞—Ç–æ–≤
        print("\n" + "=" * 50)
        print("–†–ï–ó–£–õ–¨–¢–ê–¢–´ –°–ö–ê–ù–ò–†–û–í–ê–ù–ò–Ø")
        print("=" * 50)
        
        if self.open_ports:
            self.open_ports.sort()
            for port in self.open_ports:
                service = self.get_service_name(port)
                print(f"–ü–æ—Ä—Ç {port}/tcp : –û–¢–ö–†–´–¢ - {service}")
        else:
            print("–û—Ç–∫—Ä—ã—Ç—ã–µ –ø–æ—Ä—Ç—ã –Ω–µ –Ω–∞–π–¥–µ–Ω—ã")
        
        print(f"\n–í—Ä–µ–º—è —Å–∫–∞–Ω–∏—Ä–æ–≤–∞–Ω–∏—è: {end_time - start_time}")

# –ò—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏–µ —Å–∫–∞–Ω–µ—Ä–∞

# network_analyzer.py
import socket
import subprocess
import platform
import re
from datetime import datetime

class NetworkAnalyzer:
    def __init__(self):
        self.system_info = {}
        self.network_info = {}
    
    def get_system_info(self):
        """–ü–æ–ª—É—á–µ–Ω–∏–µ –∏–Ω—Ñ–æ—Ä–º–∞—Ü–∏–∏ –æ —Å–∏—Å—Ç–µ–º–µ"""
        self.system_info = {
            'hostname': socket.gethostname(),
            'os': platform.system() + " " + platform.release(),
            'architecture': platform.architecture()[0]
        }
        return self.system_info
    
    def get_ip_info(self):
        """–ü–æ–ª—É—á–µ–Ω–∏–µ IP –∏–Ω—Ñ–æ—Ä–º–∞—Ü–∏–∏"""
        try:
            # –ü–æ–ª—É—á–∞–µ–º –ª–æ–∫–∞–ª—å–Ω—ã–π IP
            hostname = socket.gethostname()
            local_ip = socket.gethostbyname(hostname)
            
            # –ü–æ–ª—É—á–∞–µ–º –≤–Ω–µ—à–Ω–∏–π IP (—á–µ—Ä–µ–∑ –≤–Ω–µ—à–Ω–∏–π —Å–µ—Ä–≤–∏—Å)
            external_ip = self.get_external_ip()
            
            self.network_info['local_ip'] = local_ip
            self.network_info['external_ip'] = external_ip
            
            return {
                'local_ip': local_ip,
                'external_ip': external_ip
            }
        except Exception as e:
            return {'error': str(e)}
    
    def get_external_ip(self):
        """–ü–æ–ª—É—á–µ–Ω–∏–µ –≤–Ω–µ—à–Ω–µ–≥–æ IP –∞–¥—Ä–µ—Å–∞"""
        try:
            with socket.socket(socket.AF_INET, socket.SOCK_DGRAM) as s:
                s.connect(("8.8.8.8", 80))
                return s.getsockname()[0]
        except:
            try:
                import urllib.request
                return urllib.request.urlopen('https://api.ipify.org').read().decode('utf8')
            except:
                return "–ù–µ —É–¥–∞–ª–æ—Å—å –æ–ø—Ä–µ–¥–µ–ª–∏—Ç—å"
    
    def scan_local_network(self, network_range="192.168.1.0/24"):
        """–°–∫–∞–Ω–∏—Ä–æ–≤–∞–Ω–∏–µ –ª–æ–∫–∞–ª—å–Ω–æ–π —Å–µ—Ç–∏"""
        print(f"–°–∫–∞–Ω–∏—Ä–æ–≤–∞–Ω–∏–µ —Å–µ—Ç–∏: {network_range}")

        for i in range(1, 255):
            ip = base_ip + str(i)
            try:
                # –ü–∏–Ω–≥ —Ö–æ—Å—Ç–∞
                param = '-n' if platform.system().lower() == 'windows' else '-c'
                command = ['ping', param, '1', '-w', '1000', ip]
                
                result = subprocess.run(command, capture_output=True, text=True)
                
                if "TTL=" in result.stdout or "ttl=" in result.stdout:
                    active_hosts.append(ip)
                    print(f"–ù–∞–π–¥–µ–Ω –∞–∫—Ç–∏–≤–Ω—ã–π —Ö–æ—Å—Ç: {ip}")
                    
                    # –ü–æ–ø—ã—Ç–∫–∞ –ø–æ–ª—É—á–∏—Ç—å –∏–º—è —Ö–æ—Å—Ç–∞
                    try:
                        hostname = socket.gethostbyaddr(ip)[0]
                        print(f"  –ò–º—è —Ö–æ—Å—Ç–∞: {hostname}")
                    except:
                        print(f"  –ò–º—è —Ö–æ—Å—Ç–∞: –Ω–µ —É–¥–∞–ª–æ—Å—å –æ–ø—Ä–µ–¥–µ–ª–∏—Ç—å")
                    
            except Exception as e:
                pass
        
        return active_hosts
    
    def get_network_connections(self):
        """–ü–æ–ª—É—á–µ–Ω–∏–µ –∞–∫—Ç–∏–≤–Ω—ã—Ö —Å–µ—Ç–µ–≤—ã—Ö –ø–æ–¥–∫–ª—é—á–µ–Ω–∏–π"""

            else:

        except Exception as e:
